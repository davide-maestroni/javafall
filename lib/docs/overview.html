Parallel programming on the go.

Usage examples

The code below shows how it is possible to implement the computation of the root mean square of some input data, by distributing it in a number of background threads.

Example 1: via invocation implementations: 
    
        public class Main {

            public static void main(final String[] args) {

                final Routine<String, Integer> parseRoutine = on(tokenOf(Parse.class)).buildRoutine();
                final Routine<Integer, Integer> sumRoutine = on(tokenOf(Sum.class)).buildRoutine();
                final Routine<Integer, Integer> squareRoutine = on(tokenOf(Square.class)).buildRoutine();

                final Integer rms =
                        sumRoutine.callAsync(squareRoutine.callParallel(parseRoutine.callParallel(args)))
                                  .readFirst();

                System.out.println(rms);

                System.exit(0);
            }

            private static class Parse extends TemplateInvocation<String, Integer> {

                @Override
                public void onInput(final String s, @Nonnull final ResultChannel<Integer> result) {

                    result.pass(Integer.parseInt(s));
                }
            }

            private static class Square extends TemplateInvocation<Integer, Integer> {

                @Override
                public void onInput(final Integer integer, @Nonnull final ResultChannel<Integer> result) {

                    final int input = integer;
                    result.pass(input * input);
                }
            }

            private static class Sum extends TemplateInvocation<Integer, Integer> {

                private int mSum;

                @Override
                public void onInit() {

                    mSum = 0;
                }

                @Override
                public void onInput(final Integer integer, @Nonnull final ResultChannel<Integer> result) {

                    mSum += integer;
                }

                @Override
                public void onResult(@Nonnull final ResultChannel<Integer> result) {

                    result.pass(mSum);
                }
            }
        }
Example 2: via reflection: 
    
        @Async(lockName = Async.NULL_LOCK)
        public class Main {

            private static final String PARSE = "parse";

            private static final String SQUARE = "square";

            private static final String SUM = "sum";

            public static void main(final String[] args) {

                final Routine<Object, Object> parseRoutine = on(Main.class).asyncMethod(PARSE);
                final Routine<Object, Object> squareRoutine = on(Main.class).asyncMethod(SQUARE);
                final Routine<Object, Object> sumRoutine = on(Main.class).asyncMethod(SUM);

                final Integer rms = (Integer) sumRoutine.callAsync(
                        (Object) squareRoutine.callParallel(parseRoutine.callParallel(args)).readAll())
                                                        .readFirst();

                System.out.println(rms);

                System.exit(0);
            }

            @Async(PARSE)
            private static int parseArg(final String arg) {

                return Integer.parseInt(arg);
            }

            @Async(SQUARE)
            private static int square(final int value) {

                return value * value;
            }

            @Async(SUM)
            private static int sum(final List<Integer> values) {

                int sum = 0;

                for (final int value : values) {

                    sum += value;
                }

                return sum;
            }
        }
Example 3: via proxy: 
    
        public class Main {

            public static void main(final String[] args) {

                final AsyncMain proxy = on(new Main()).proxy(AsyncMain.class);
                final int rms = proxy.sum(proxy.square(proxy.parseArg(args)));

                System.out.println(rms);

                System.exit(0);
            }

            private int parseArg(final String arg) {

                return Integer.parseInt(arg);
            }

            private int square(final int value) {

                return value * value;
            }

            private int sum(final int... values) {

                int sum = 0;

                for (final int value : values) {

                    sum += value;
                }

                return sum;
            }

            @Async(lockName = Async.NULL_LOCK)
            private interface AsyncMain {

                @AsyncType(int.class)
                public OutputChannel<Integer> parseArg(@ParallelType(String.class) String[] args);

                @AsyncType(int.class)
                public int[] square(@ParallelType(int.class) OutputChannel<Integer> value);

                public int sum(int... values);
            }
        }
Example 4: via code generation: 
    
        public class Main {

            public static void main(final String[] args) {

                final Async proxy = on(new Main()).asClass(Async.class);
                final int rms = proxy.sum(proxy.square(proxy.parseArg(args)));

                System.out.println(rms);

                System.exit(0);
            }

            public int parseArg(final String arg) {

                return Integer.parseInt(arg);
            }

            public int square(final int value) {

                return value * value;
            }

            public int sum(final int... values) {

                int sum = 0;

                for (final int value : values) {

                    sum += value;
                }

                return sum;
            }

            @AsyncClass(Main.class)
            @Async(lockName = Async.NULL_LOCK)
            private interface Async {

                @AsyncType(int.class)
                public OutputChannel<Integer> parseArg(@ParallelType(String.class) String[] args);

                @AsyncType(int.class)
                public int[] square(@ParallelType(int.class) OutputChannel<Integer> value);

                public int sum(int... values);
            }
