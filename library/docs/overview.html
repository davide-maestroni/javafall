Usage examples

Example 1: Root mean square:

The code below shows how it is possible to implement the computation of the root mean square of some input data, by distributing it in a number of background threads. 

    
        public class Main {

            public static void main(final String[] args) {

                final Routine<String, Integer> parseRoutine = on(tokenOf(Parse.class)).buildRoutine();
                final Routine<Integer, Integer> sumRoutine = on(tokenOf(Sum.class)).buildRoutine();
                final Routine<Integer, Integer> squareRoutine = on(tokenOf(Square.class)).buildRoutine();

                final Integer rms =
                        sumRoutine.runAsync(squareRoutine.runParallel(parseRoutine.runParallel(args)))
                                  .readFirst();

                System.out.println(rms);

                System.exit(0);
            }

            private static class Parse extends BasicExecution<String, Integer> {

                @Override
                public void onInput(@Nullable final String s,
                        @NonNull final ResultChannel<Integer> results) {

                    results.pass(Integer.parseInt(s));
                }
            }

            private static class Square extends BasicExecution<Integer, Integer> {

                @Override
                public void onInput(final Integer integer, @NonNull final ResultChannel<Integer> results) {

                    final int input = integer;
                    results.pass(input * input);
                }
            }

            private static class Sum extends BasicExecution<Integer, Integer> {

                private int mSum;

                @Override
                public void onInit() {

                    mSum = 0;
                }

                @Override
                public void onInput(final Integer integer, @NonNull final ResultChannel<Integer> results) {

                    mSum += integer;
                }

                @Override
                public void onResult(@NonNull final ResultChannel<Integer> results) {

                    results.pass(mSum);
                }
            }
        }
